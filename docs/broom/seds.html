 15def _get_CMB_SED(frequencies: List[float], units: str = "uK_CMB",
 16    bandwidths: Optional[Union[List[str], List[float]]] = None
 17) -> np.ndarray:
 18    """
 19    Function to compute the SED of the CMB in different units.
 20
 21    Parameters
 22    ----------
 23        frequencies : list
 24            Frequencies in GHz.
 25        units : str, optional
 26            Units of the output SED. Options are "uK_CMB", "uK_RJ" and all their multiples (e.g., "mK_CMB", "mK_RJ") or "Jy_sr".
 27            Default is "uK_CMB".
 28        bandwidths : list or None, optional
 29            If list of str, it should contain the paths to npy files with frequency bands and response.
 30            If list of floats, it should contain the relative bandwidths in GHz for each frequency channel.
 31            If None, the SED is computed at the frequencies without assuming any bandwidth.
 32
 33    Returns
 34    -------
 35        np.ndarray
 36            SED of the CMB in the specified units.
 37    """
 38
 39    if units[-3:] == "CMB":
 40        return (u.Unit(units)).to(u.uK_CMB) * np.ones(len(frequencies))
 41
 42    if bandwidths is not None:
 43        sed = []
 44        for idx_freq, frequency in enumerate(frequencies):
 45            if isinstance(bandwidths[0], str):
 46                freqs_band, weights = np.load(bandwidths[idx_freq])
 47            else:
 48                freq_min = frequency * (1 - ( bandwidths[idx_freq] / 2 ))
 49                freq_max = frequency * (1 + ( bandwidths[idx_freq] / 2 ))
 50                steps = int(freq_max - freq_min + 1)
 51                freqs_band = np.linspace(freq_min, freq_max, steps)
 52                weights = np.ones_like(freqs_band) # The tophat is defined in intensity units (Jy/sr)
 53            weights_rj = normalize_weights(freqs_band, weights)
 54
 55            sed_band = np.zeros(1)
 56            for i, (freq, _weight) in enumerate(zip(freqs_band, weights_rj)):
 57                sed_ = u.uK_CMB.to(u.uK_RJ, equivalencies=u.cmb_equivalencies(freq * u.GHz))
 58                trapz_step_inplace(freqs_band, weights_rj, i, sed_, sed_band)
 59
 60            if units != "Jy_sr":
 61                sed.append((sed_band * bandpass_unit_conversion(freqs_band * u.GHz, weights, getattr(u, units))).value[0])
 62            else:
 63                sed.append((sed_band * bandpass_unit_conversion(freqs_band * u.GHz, weights, u.Jy / u.sr)).value[0])
 64
 65        return np.array(sed)
 66
 67    else:
 68        if units[-2:] == "RJ":
 69            return u.uK_CMB.to(u.Unit(units), equivalencies=u.cmb_equivalencies(frequencies * u.GHz))
 70        elif units == "Jy_sr":
 71            return u.uK_CMB.to((u.Jy / u.sr), equivalencies=u.cmb_equivalencies(frequencies * u.GHz))

 73def _get_moments_SED(
 74    frequencies: List[float],
 75    moms_list: List[str],
 76    beta_d: float = 1.54,
 77    T_d: float = 20.0,
 78    beta_s: float = -3.0,
 79    nu_ref_d: float = 353.0,
 80    nu_ref_s: float = 30.0,
 81    units: str = "uK_CMB",
 82    bandwidths: Optional[Union[List[str], List[float]]] = None
 83) -> np.ndarray:
 84    """
 85    Function to compute the SED of the moments in different units.
 86
 87    Parameters
 88    ----------
 89        frequencies : list
 90            Frequencies observed in GHz.
 91        moms_list : list
 92            List of moments to compute. It can include
 93            - '0d': zeroth moment of modified black body (thermal dust)
 94            - '0s': zeroth moment of power law (synchrotron)
 95            - '1bd': first derivative of modified black body wrt beta
 96            - '1bs': first derivative of power law wrt beta
 97            - '1Td': first derivative of modified black body wrt temperature
 98            - '2bd': second derivative of modified black body wrt beta
 99            - '2bs': second derivative of power law wrt beta
100            - '2Td': second derivative of modified black body wrt temperature
101            - '2bdTd': second derivative of modified black body wrt beta and temperature
102            - '2Tdbd': second derivative of modified black body wrt temperature and beta
103        beta_d : float, optional
104            Spectral index of the modified black body (thermal dust). It refers to muK_RJ units SED.
105            Default is 1.54.
106        T_d : float, optional
107            Temperature of the modified black body (thermal dust) in Kelvin. It refers to muK_RJ units SED. 
108            Default is 20.
109        beta_s : float, optional
110            Spectral index of the power law (synchrotron). It refers to muK_RJ units SED. 
111            Default is -3.
112        nu_ref_d : float, optional
113            Reference frequency for the modified black body (thermal dust) in GHz. 
114            Default is 353.
115        nu_ref_s : float, optional
116            Reference frequency for the power law (synchrotron) in GHz.
117            Default is 30.
118        units : str, optional
119            Units of the output SED. Options are "uK_CMB", "uK_RJ", or "Jy_sr" and all their multiples (e.g., "mK_CMB", "mK_RJ", "mJy_sr").
120        bandwidths : list or None, optional
121            If list of str, it should contain the paths to npy files with frequency bands and response.
122            If list of floats, it should contain the relative bandwidths in GHz for each frequency channel.
123            If None, the SED is computed at the frequencies without assuming any bandwidth.
124    
125    Returns
126    -------
127        np.ndarray
128            SED of the moments in the specified units. The shape is (len(moms_list), len(frequencies)).
129    """
130    
131    moments_funcs = {
132    '0d': _mbb_0,
133    '0s': _pl_0,
134    '1bd': _mbb_1b,
135    '1bs': _pl_1b,
136    '1Td': _mbb_1T,
137    '2bd': _mbb_2b,
138    '2bs': _pl_2b,
139    '2Td': _mbb_2T,
140    '2bdTd': _mbb_2bT,
141    '2Tdbd': _mbb_2bT,
142    }
143
144    all_seds = []
145    for mom in moms_list:
146        sed = []
147
148        for idx_freq, frequency in enumerate(frequencies):
149            if bandwidths is not None:
150                if isinstance(bandwidths[0], str):
151                    freqs_band, weights = np.load(bandwidths[idx_freq])
152                else:
153                    freq_min = frequency * (1 - ( bandwidths[idx_freq] / 2 ))
154                    freq_max = frequency * (1 + ( bandwidths[idx_freq] / 2 ))
155                    steps = int(freq_max - freq_min + 1)
156                    freqs_band = np.linspace(freq_min, freq_max, steps)
157                    weights = np.ones_like(freqs_band) # The tophat is defined in intensity units (Jy/sr)
158                weights_rj = normalize_weights(freqs_band, weights)
159
160                sed_band = np.zeros(1)
161                for i, (freq, _weight) in enumerate(zip(freqs_band, weights_rj)):
162                    if 'd' in mom:
163                        sed_ = moments_funcs[mom](freq, beta_d, T_d, nu_ref = nu_ref_d)
164                    elif 's' in mom:
165                        sed_ = moments_funcs[mom](freq, beta_s, nu_ref = nu_ref_s)
166                    trapz_step_inplace(freqs_band, weights_rj, i, sed_, sed_band)          
167            else:
168                if 'd' in mom:
169                    sed_band = moments_funcs[mom](frequency, beta_d, T_d, nu_ref = nu_ref_d)
170                elif 's' in mom:
171                    sed_band = moments_funcs[mom](frequency, beta_s, nu_ref = nu_ref_s)
172                freqs_band = frequency
173                weights = None
174
175            if units != "Jy_sr":
176                if units[-3:] == "CMB":
177                    sed.append((sed_band * bandpass_unit_conversion(freqs_band * u.GHz, weights, getattr(u, units))).value[0])
178                elif units[-2:] == "RJ":
179                    sed.append((sed_band * bandpass_unit_conversion(freqs_band * u.GHz, weights, getattr(u, units))).value)
180            else:
181                sed.append((sed_band * bandpass_unit_conversion(freqs_band * u.GHz, weights, u.Jy / u.sr)).value[0])
182
183        all_seds.append(sed)
184    
185    return np.array(all_seds)

188def _mbb_0(nu, beta, T, nu_ref = 335.):
189    """
190    Function to compute the SED of a modified black body.
191
192    Parameters
193    ----------
194        nu : float or np.ndarray
195            Frequency in GHz.
196        beta : float
197            Spectral index of the modified black body.
198        T : float
199            Temperature in Kelvin.
200        nu_ref : float, optional
201            Reference frequency in GHz. Default is 335 GHz.
202    
203    Returns
204    -------
205        np.ndarray
206            SED of the modified black body in CMB brightness units.
207    """
208    x = (h * nu * (1e+9)) / (k * T)
209    return ((nu/nu_ref)**(beta+1)) / (np.exp(x)-1)

211def _mbb_1b(nu, beta, T, nu_ref = 335.):
212    """
213    Function to compute the SED of first derivative of modified black body wrt beta.
214
215    Parameters
216    ----------
217        nu : float or np.ndarray
218            Frequency in GHz.
219        beta : float
220            Spectral index of the modified black body.
221        T : float
222            Temperature in Kelvin.
223        nu_ref : float, optional
224            Reference frequency in GHz. Default is 335 GHz.
225        
226    Returns
227    -------
228        np.ndarray
229            SED of the first derivative of modified black body wrt beta in CMB brightness units.
230    """
231    return np.log(nu/nu_ref) * _mbb_0(nu, beta, T, nu_ref = nu_ref)

233def _mbb_1T(nu, beta, T, nu_ref = 335.):
234    """
235    Function to compute the SED of first derivative of modified black body wrt temperature.
236
237    Parameters
238    ----------
239        nu : float or np.ndarray
240            Frequency in GHz.
241        beta : float
242            Spectral index of the modified black body.
243        T : float
244            Temperature in Kelvin.
245        nu_ref : float, optional
246            Reference frequency in GHz. Default is 335 GHz.
247    
248    Returns
249    -------
250        np.ndarray
251            SED of the first derivative of modified black body wrt temperature in CMB brightness units.
252    """
253    x = (h * nu * (1e+9)) / (k * T)
254    return x * np.exp(x) * _mbb_0(nu, beta, T, nu_ref = nu_ref) / (T * (np.exp(x)-1))

278def _mbb_2T(nu, beta, T, nu_ref = 335.):
279    """
280    Function to compute the SED of second derivative of modified black body wrt temperature.
281
282    Parameters
283    ----------
284        nu : float or np.ndarray
285            Frequency in GHz.
286        beta : float
287            Spectral index of the modified black body.
288        T : float
289            Temperature in Kelvin.
290        nu_ref : float, optional

292
293    Returns

295        np.ndarray
296            SED of the second derivative of modified black body wrt temperature in CMB brightness units.
297    """
298    x = (h * nu * (1e+9)) / (k * T)
299    return ((x/np.tanh(x/2)) - 2.) * _mbb_1T(nu, beta, T, nu_ref = nu_ref) / T

402            nls_number = _get_needlet_windows_(compsep_run["needlet_config"], lmax).shape[0]
403        elif compsep_run["method"] in ["c_ilc", "c_pilc"]:
