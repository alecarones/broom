 52def _get_data_foregrounds_(config: Configs, **kwargs: Any) -> SimpleNamespace:
 53    """
 54    Load or generate foreground maps based on configuration.
 55
 56    Parameters
 57    ----------
 58        config: Configs
 59            Configuration parameters. It should have the following attributes:
 60            - `generate_input_foregrounds`: Whether to generate foreground maps.
 61            - `foreground_models`: List of foreground models to generate.
 62            - 'instrument': a dictionary containing the instrument configuration, including:
 63                - `frequency`: List of instrument frequencies in GHz.
 64                - `beams`: Type of beams to be used (e.g., "gaussian", "file_l", "file_lm").
 65                - `fwhm`: List of full width at half maximum (FWHM) for each frequency channel in arcmin. Used if beams are "gaussian".
 66                - 'depth_I': Depth for intensity maps in arcmin*uK_CMB (optional). 
 67                            If not provided, it will be assumed to be the polarization depth divided by sqrt(2).
 68                            Used if path_depth_maps is not provided.
 69                - 'depth_P': Depth for polarization maps in arcmin*uK_CMB (optional).
 70                            If not provided, it will be assumed to be the intensity depth multiplied by sqrt(2).
 71                            Used if path_depth_maps is not provided.
 72                - `path_beams`: Full path to the beams files (if using "file_l" or "file_lm" beams). 
 73                            The code will look for files named "{path_beams}_{channel_tag}.fits" for each frequency channel.
 74                - `channels_tags`: List of tags for each frequency channel, used for loading beams, bandpasses or depth maps.
 75                - 'bandwidths': List of relative bandwidths for each frequency channel (optional, used if bandpass_integrate is True).
 76                            Used if path_bandpasses is not provided.
 77                - `path_depth_maps`: Full path to depth maps (optional, used if generating noise). 
 78                            The code will look for files named "{path_depth_maps}_{channel_tag}.fits" for each frequency channel.
 79                - `path_hits_maps`: Full path to hits maps (optional, used if generating noise and 'depth_maps' is not provided).
 80                            If it does not end with .fits, the code will look for files named "{path_hits_maps}_{channel_tag}.fits" for each frequency channel.
 81                - `path_bandpasses`: Full path to bandpass files (optional, used if bandpass_integrate is True).
 82                            It will look for files named as "{path_bandpasses}_{channel_tag}.npy" for each channel tag.
 83                            Each file should be a 2D array which has the first column a list of frequencies in GHz and the second column the corresponding bandpass response.
 84                - `ell_knee`: Lists of knee frequencies for each channel for the noise power spectrum (optional).
 85                            If it is a single list it will be applied to temperature only.
 86                            If it is a list of two lists it will be applied to temperature (first list) and polarization (second list).
 87                            If not provided, white noise is assumed.
 88                - `alpha_knee`: List of spectral indices of the noise power spectrum for each channel (optional).
 89                            If not provided, white noise is assumed.
 90            - `nside`: HEALPix resolution.
 91            - `lmax`: Maximum multipole for the simulation.
 92            - `return_fgd_components`: Whether to return individual foreground components.
 93            - `fgds_path`: Path where saving or loading foreground maps.
 94            - `save_inputs`: Whether to save generated foreground maps to disk.        
 95            - `pixel_window_in`: Whether to apply pixel window smoothing.
 96            - `units`: Units for the foreground maps (e.g., 'uK_CMB').
 97            - `data_type`: Type of data to return, either "maps" or "alms".
 98            - `bandpass_integrate`: Whether to integrate foreground components across bandpasses.
 99            - `lmin`: Minimum multipole to keep in the simulation.
100            - `coordinates`: Coordinate system for the maps (e.g., "G" for Galactic).
101        kwargs: dict, optional
102            Additional keyword arguments forwarded to alm computation.
103
104    Returns
105    -------
106        SimpleNamespace
107            Foregrounds object containing single components (optionally) and total map.
108    """

110
111    if config.generate_input_foregrounds:
112        if config.verbose:
113            msg = f"Generating foreground maps of {''.join(config.foreground_models)} model"
114            if config.bandpass_integrate:
115                msg += " with bandpass integration"

117
118        foregrounds = _get_foregrounds(
119            config.foreground_models,
120            config.instrument,
121            config.nside,
122            config.nside_in,
123            config.lmax,
124            return_components=config.return_fgd_components,
125            pixel_window=config.pixel_window_in,
126            units=config.units,
127            return_alms=(config.data_type == "alms"),

129            lmin=config.lmin,
130            coordinates=config.coordinates,

143        foregrounds.total = _load_input_foregrounds(config.fgds_path, "".join(config.foreground_models))

309def _save_input_foregrounds(fgds_path: str, foregrounds: SimpleNamespace, foreground_models: List[str]) -> None:
310    """
311    Save foreground maps for all models and the total.
312    Assumes 'foregrounds' is a SimpleNamespace or similar with components as attributes.
313
314    Parameters
315    ----------
316        fgds_path: str
317            Path where saving foreground maps, without extension.
318        foregrounds: SimpleNamespace
319            Foreground components, should have attributes for each model and a 'total' attribute.
320        foreground_models: List[str]
321            List of foreground model names to save. Each name should match an attribute in 'foregrounds'.
322
323    Returns
324    -------
325        None    
326    """
327    os.makedirs(os.path.dirname(fgds_path), exist_ok=True)
328    # Save total foreground map
329    np.save(fgds_path + f'_{"".join(foreground_models)}', foregrounds.total)
330
331    if len(vars(foregrounds)) > 1:
332        fg_attrs = {k: v for k, v in vars(foregrounds).items() if k != "total"}
333        if len(fg_attrs) != len(foreground_models):
334            raise ValueError(
335                f"Number of foreground components ({len(fg_attrs)}) does not match number of models ({len(foreground_models)}).")
336
337        for fmodel in foreground_models:
338            # Try matching longest prefix first (3,2,1)
339            attr = None
340            for length in (3, 2, 1):
341                if len(fmodel) > length:
342                    prefix = fmodel[:length]
343                    attr = prefix_to_attr.get(prefix)
344                    if attr is not None:
345                        break
346            if attr is None:
347                raise ValueError(f"Unknown foreground model prefix for '{fmodel}'")
348            if attr not in fg_attrs:
349                raise ValueError(f"Foreground attribute '{attr}' missing in foregrounds object")
350
351            np.save(fgds_path + f"_{fmodel}", fg_attrs[attr])

373def _get_noise_simulation(config: Configs, nsim: Optional[Union[int, str]] = None, **kwargs) -> np.ndarray:
374    """
375    Generate noise simulations for the instrument frequencies.
376
377    Parameters
378    ----------
379        config: Configs
380            Configuration parameters including instrument settings. It should have the following attributes:
381            - `instrument.frequency`: List of instrument frequencies.
382            - `instrument.depth_I`: Depth for intensity maps.
383            - `instrument.depth_P`: Depth for polarization maps.
384            - `instrument.path_depth_maps`: Path to depth maps (optional).
385            - `instrument.path_hits_maps`: Path to hits maps (optional).
386            - `nside`: HEALPix resolution.
387            - `lmax`: Maximum multipole for the simulation.
388            - `data_type`: Type of data to return, either "maps" or "alms".
389            - `units`: Units for the noise maps (e.g., 'uK_CMB').
390            - `lmin`: Minimum multipole to keep in the simulation.
391            - `seed_noise`: Seed for noise generation (optional).    
392        nsim: int or str, optional
393            Simulation index to save the maps and vary the random seed (optional). Default: None.
394        kwargs: dict, optional
395            Additional keyword arguments for `hp.map2alm`.
396
397    Returns
398    -------
399        noise: np.ndarray
400            array of noise maps or alms. Shape is (n_freq, 3, n_pix) for maps or (n_freq, 3, n_alm) for alms.
401    """
402
403    if nsim is not None:
404        if not isinstance(nsim, (int, str)):
405            raise ValueError("nsim must be an integer or a string.")
406        if isinstance(nsim, int):
407            nsim = str(nsim).zfill(5)
408
409
410    # Setup seed for reproducibility
411    if config.seed_noise is None:
412        seed = None
413    else:
414        if nsim is not None:
415            seed = config.seed_noise + int(nsim) * 3 * len(config.instrument.frequency)
416        else:
417            seed = config.seed_noise    
418
419    if not hasattr(config.instrument, 'path_depth_maps'):
420        if not hasattr(config.instrument, 'depth_I') and not hasattr(config.instrument, 'depth_P'):
421            raise ValueError('Provided instrumental setting must have either depth_I or depth_P attributes.')
422        elif not hasattr(config.instrument, 'depth_I') and hasattr(config.instrument, 'depth_P'):
423            config.instrument.depth_I = config.instrument.depth_P / np.sqrt(2)
424            _log('Warning: No intensity map depth provided. Assuming it to be the polarization one divided by sqrt(2).', verbose=config.verbose)
425        elif not hasattr(config.instrument, 'depth_P') and hasattr(config.instrument, 'depth_I'):
426            config.instrument.depth_P = config.instrument.depth_I * np.sqrt(2)
427            _log('Warning: No polarization map depth provided. Assuming it to be the intensity one multiplied by sqrt(2).', verbose=config.verbose)
428        depth_i = config.instrument.depth_I
429        depth_p = config.instrument.depth_P
430
431        # Load or set hits map if available
432        if hasattr(config.instrument, 'path_hits_maps'):
433            if config.instrument.path_hits_maps.endswith(".fits"):
434                hits_map = hp.read_map(config.instrument.path_hits_maps, field=0, dtype=np.float64)
435                if hp.get_nside(hits_map) != config.nside:
436                    hits_map = hp.ud_grade(hits_map, nside_out=config.nside, power=-2)
437                hits_map /= np.amax(hits_map)
438    else:
439        depth_i = [1.] * len(config.instrument.frequency)
440        depth_p = [1.] * len(config.instrument.frequency)
441        
442    # Convert depths to requested units with CMB equivalencies
443    #depth_i *= u.arcmin * u.uK_CMB
444    #depth_i = depth_i.to(getattr(u, config.units) * u.arcmin, equivalencies=u.cmb_equivalencies(config.instrument.frequency * u.GHz))
445    #depth_p *= u.arcmin * u.uK_CMB
446    #depth_p = depth_p.to(getattr(u, config.units) * u.arcmin, equivalencies=u.cmb_equivalencies(config.instrument.frequency * u.GHz))
447    bandwidths = _get_bandwidths(config, np.arange(len(config.instrument.frequency)))
448    A_cmb = _get_CMB_SED(config.instrument.frequency, units=config.units, bandwidths=bandwidths)
449    depth_i = depth_i * A_cmb
450    depth_p = depth_p * A_cmb
451
452    # Precompute conversion factor from arcmin to radians
453    acm_to_rad = (np.pi / (180 * 60)) 
454
455    # Get ell filter if needed
456    fell = _get_ell_filter(config.lmin, config.lmax) if config.lmin > 2 else None
457
458    noise = []
459    for nf, _ in enumerate(config.instrument.frequency):
460        # Load depth maps if path provided
461        if hasattr(config.instrument, 'path_depth_maps'):
462            depth_map_fn = config.instrument.path_depth_maps + f"_{config.instrument.channels_tags[nf]}.fits"
463            try:
464                depth_maps_in = hp.read_map(depth_map_fn, field=(0,1), dtype=np.float64)
465            except IndexError:
466                print("Warning: Unable to read depth maps from the provided path for I and P, provided depth map is assumed to refer to polarization.")
467                depth_maps_in = hp.read_map(depth_map_fn, field=0, dtype=np.float64)
468                depth_maps_in = np.array([depth_maps_in / np.sqrt(2), depth_maps_in])
469            if hp.get_nside(depth_maps_in[0]) != config.nside:
470                depth_maps = np.array(
471                    [np.sqrt(hp.ud_grade(dm**2, nside_out=config.nside, power=2)) for dm in depth_maps_in])
472            else:
473                depth_maps = np.copy(depth_maps_in)
474            del depth_maps_in
475        elif hasattr(config.instrument, 'path_hits_maps'):
476            if not config.instrument.path_hits_maps.endswith(".fits"):
477                hits_file = config.instrument.path_hits_maps + f"_{config.instrument.channels_tags[nf]}.fits"
478                hits_map = hp.read_map(hits_file, field=0, dtype=np.float64)
479                if hp.get_nside(hits_map) != config.nside:
480                    hits_map = hp.ud_grade(hits_map, nside_out=config.nside, power=-2)
481                hits_map /= np.amax(hits_map)
482
483        if seed is not None:
484            np.random.seed(seed + (nf * 3))
485        # Generate noise power spectra
486#        N_ell_T = (depth_i.value[nf] * acm_to_rad) ** 2 * np.ones(config.lmax + 1)
487#        N_ell_P = (depth_p.value[nf] * acm_to_rad) ** 2 * np.ones(config.lmax + 1)
488        N_ell_T = (depth_i[nf] * acm_to_rad) ** 2 * np.ones(config.lmax + 1)
489        N_ell_P = (depth_p[nf] * acm_to_rad) ** 2 * np.ones(config.lmax + 1)
490        N_ell = np.array([N_ell_T, N_ell_P, N_ell_P, 0.*N_ell_P])
491
492        # Add knee frequency noise if provided
493        if hasattr(config.instrument, 'ell_knee') and hasattr(config.instrument, 'alpha_knee'):
494            ell = np.arange(config.lmax + 1)
495            if isinstance(config.instrument.alpha_knee, list) and isinstance(config.instrument.ell_knee, list):
496                if np.array(config.instrument.alpha_knee).ndim == 2 and np.array(config.instrument.ell_knee).ndim == 2:
497                    if len(config.instrument.alpha_knee[0]) != len(config.instrument.ell_knee[0]) or len(config.instrument.alpha_knee[1]) != len(config.instrument.ell_knee[1]):
498                        raise ValueError('alpha_knee and ell_knee must have the same length.')
499                    if (len(config.instrument.ell_knee[0]) != len(config.instrument.frequency)) or (len(config.instrument.ell_knee[1]) != len(config.instrument.frequency)):
500                        raise ValueError('alpha_knee and ell_knee must have the same length as the number of frequencies.')
501                    N_ell[0] *= (1 + (ell / config.instrument.ell_knee[0][nf]) ** config.instrument.alpha_knee[0][nf])
502                    N_ell[1:] *= (1 + (ell / config.instrument.ell_knee[1][nf]) ** config.instrument.alpha_knee[1][nf])
503                elif np.array(config.instrument.alpha_knee).ndim == 1 and np.array(config.instrument.ell_knee).ndim == 1:
504                    if len(config.instrument.alpha_knee) != len(config.instrument.frequency) or len(config.instrument.ell_knee) != len(config.instrument.frequency):
505                        raise ValueError('alpha_knee and ell_knee must have the same length as the number of frequencies.')
506                    N_ell[0] *= (1 + (ell / config.instrument.ell_knee[nf]) ** config.instrument.alpha_knee[nf])
507                elif np.array(config.instrument.alpha_knee).ndim == 1 and np.array(config.instrument.ell_knee).ndim == 2:
508                    if len(config.instrument.alpha_knee) != len(config.instrument.frequency):
509                        raise ValueError('alpha_knee must have the same length as the number of frequencies.')
510                    if len(config.instrument.ell_knee[0]) != len(config.instrument.frequency) or len(config.instrument.ell_knee[1]) != len(config.instrument.frequency):
511                        raise ValueError('ell_knee lists must have the same length as the number of frequencies.')
512                    N_ell[0] *= (1 + (ell / config.instrument.ell_knee[0][nf]) ** config.instrument.alpha_knee[nf])
513                    N_ell[1:] *= (1 + (ell / config.instrument.ell_knee[1][nf]) ** config.instrument.alpha_knee[nf])
514            else:
515                raise ValueError('alpha_knee and ell_knee must be both lists or lists of 2 lists')
516
517        # Generate noise alm
518        alm_noise = hp.synalm(N_ell, lmax=config.lmax, new=True)
519
520        # Apply ell filter if applicable
521        if fell is not None:
522            for f in range(3):
523                alm_noise[f] = hp.almxfl(alm_noise[f], fell)
524
525        # Generate noise maps or alms depending on data_type
526        if config.data_type=="alms":
527            if hasattr(config.instrument, 'path_depth_maps'):
528                noise_map = hp.alm2map(alm_noise, config.nside, lmax=config.lmax, pol=True)  * np.array([depth_maps[0], depth_maps[1], depth_maps[1]])
529                noise.append(hp.map2alm(noise_map, lmax=config.lmax, pol=True, **kwargs))
530            elif hasattr(config.instrument, 'path_hits_maps'):
531                noise_map = hp.alm2map(alm_noise, config.nside, lmax=config.lmax, pol=True) / np.sqrt(hits_map)
532                noise_map[np.isinf(noise_map)] = 0.
533                noise.append(hp.map2alm(noise_map, lmax=config.lmax, pol=True, **kwargs))
534            else:
535                noise.append(alm_noise)
536        else:
537            if hasattr(config.instrument, 'path_depth_maps'):
538                noise.append(hp.alm2map(alm_noise, config.nside, lmax=config.lmax, pol=True) * np.array([depth_maps[0], depth_maps[1], depth_maps[1]]))
539            elif hasattr(config.instrument, 'path_hits_maps'):
540                noise_map = hp.alm2map(alm_noise, config.nside, lmax=config.lmax, pol=True) / np.sqrt(hits_map)
541                noise_map[np.isinf(noise_map)] = 0.
542                noise.append(noise_map)
543            else:
544                noise.append(hp.alm2map(alm_noise, config.nside, lmax=config.lmax, pol=True))
545
546    if config.save_inputs:
547        _save_inputs(config.noise_path, np.array(noise), nsim=nsim)
548
549    return np.array(noise)

551def _get_cmb_simulation(config: Configs, nsim: Optional[Union[int, str]] = None) -> np.ndarray:
552    """
553    Generate simulated CMB maps or alms for a given instrument configuration.
554
555    Parameters
556    ----------
557        config: Configs
558            Simulation and instrument configuration. It should have the following attributes:
559            - `lmax`: Maximum multipole for the simulation.
560            - `nside`: HEALPix resolution.
561            - `data_type`: Type of data to return, either "maps" or "alms".
562            - `cls_cmb_path`: Path to the CMB power spectrum FITS file.
563            - `seed_cmb`: Seed for CMB generation (optional).
564            - 'cls_cmb_new_ordered': Whether the new ordering of Cls is used in the CMB power spectrum FITS file.
565        nsim: int or str, optional
566            Simulation index to save the maps and vary the random seed (optional). Default: None.
567
568    Returns
569    -------
570        np.ndarray: 
571            Simulated CMB maps or harmonic coefficients (alms). Shape is (n_freq, 3, n_pix) for maps or (n_freq, 3, n_alm) for alms.
572    """
573    # Converting nsim to a string if provided
574    if nsim is not None:
575        if not isinstance(nsim, (int, str)):
576            raise ValueError("nsim must be an integer or a string.")
577        if isinstance(nsim, int):
578            nsim = str(nsim).zfill(5)
579
580    # Getting default path to CMB power spectrum, if not provided in config
581    if not config.cls_cmb_path:
582        # Define the path to the FITS file
583        raise ValueError(
584            "No CMB power spectrum path provided. Please set 'cls_cmb_path' in the configuration."
585        )
586        
587    # Load the CMB power spectrum FITS file (assumed to be in muK_CMB^2 units)
588    cls_cmb = hp.read_cl(config.cls_cmb_path)
589
590    # Initializing the seed if required
591    seed = None if not config.seed_cmb else (config.seed_cmb + int(nsim) if nsim is not None else config.seed_cmb)
592    
593    # Generating a realization of CMB alms with the loaded Cls
594    alm_cmb = _get_cmb_alms_realization(cls_cmb, config.lmax, seed = seed, new = config.cls_cmb_new_ordered)
595    
596    # Computing the high-pass filter if lmin > 2
597    fell = _get_ell_filter(config.lmin, config.lmax) if config.lmin > 2 else None
598
599    # Smoothing the CMB alms with the beams of each frequency channel
600    cmb = []
601
602    bandwidths = _get_bandwidths(config, np.arange(len(config.instrument.frequency)))
603    A_cmb = _get_CMB_SED(config.instrument.frequency, units=config.units, bandwidths=bandwidths)
604     
605    for idx, _ in enumerate(config.instrument.frequency):
606        if config.instrument.beams == "gaussian":
607            alm_cmb_i = _smooth_input_alms_(
608                alm_cmb,
609                fwhm=config.instrument.fwhm[idx],
610                nside_out=config.nside_in if config.pixel_window_in else None
611            )
612        else:
613            beamfile = config.instrument.path_beams + f"_{config.instrument.channels_tags[idx]}.fits"
614            alm_cmb_i = _smooth_input_alms_(
615                alm_cmb,
616                beam_path=beamfile,
617                symmetric_beam=(config.instrument.beams == "file_l"),
618                nside_out=config.nside_in if config.pixel_window_in else None
619            )
620
621        if fell is not None:
622            for f in range(3):
623                alm_cmb_i[f] = hp.almxfl(alm_cmb_i[f], fell)
624
625        cmb.append(A_cmb[idx] * alm_cmb_i if config.data_type == "alms" else A_cmb[idx] * hp.alm2map(
626            alm_cmb_i, config.nside, lmax=config.lmax, pol=True
627        ))
628    
629    cmb = np.array(cmb)
630
631    # Saving the CMB maps/alms if requested
632    if config.save_inputs:
633        _save_inputs(config.cmb_path, cmb, nsim=nsim)
634    return cmb

636def _get_cmb_alms_realization(
637    cls_cmb: np.ndarray,
638    lmax: int,
639    seed: Optional[int] = None,
640    new: bool = True
641) -> np.ndarray:
642    """
643    Generate a realization of CMB spherical harmonic coefficients (alms).
644
645    Parameters
646    ----------
647        cls_cmb: np.ndarray
648            Theoretical CMB angular power spectra.
649        lmax: int
650            Maximum multipole for the realization.
651        seed: Optional[int]
652            Random seed for reproducibility.
653        new: bool
654            healpy sinalm keyword which sets the assumed ordering of the Cls. Default: True.
655                    If True, use the new ordering of cl’s, ie by diagonal (e.g. TT, EE, BB, TE, EB, TB or TT, EE, BB, TE if 4 cl as input). 
656                    If False, use the old ordering, ie by row (e.g. TT, TE, TB, EE, EB, BB or TT, TE, EE, BB if 4 cl as input).
657
658    Returns
659    -------
660        np.ndarray
661            Realization of CMB alms for T, E, and B modes. Shape is (3, n_alms)
662    """
663    if seed is not None:
664        np.random.seed(seed)
665    return hp.synalm(cls_cmb, lmax=lmax, new=new)

753def _get_foreground_component(
754    instrument: dict,
755    sky: pysm3.Sky,
756    nside_out: int,
757    nside_in: int,
758    lmax: int,
759    pixel_window: bool = False,
760    bandpass_integrate: bool = False,
761    return_alms: bool = False,
762    lmin: int = 2,
763    coordinates: str = "G",
764    **kwargs
765) -> np.ndarray:
766    """
767    Generate a foreground component for each frequency channel of the instrument.
768
769    Parameters
770    ----------
771        instrument: dict
772            Instrument configuration with frequencies, fwhms (or beam paths), bandpasses or bandwidths.
773        sky: pysm3.Sky
774            PySM3 sky model for the foreground.
775        nside_out: int
776            HEALPix resolution for the output.
777        nside_in: int
778            Input HEALPix resolution. Used to apply pixel window function (if requested)
779        lmax: int
780            Maximum multipole to compute alms.
781        pixel_window: bool, optional
782            Apply pixel window smoothing if True. Default: False.
783        bandpass_integrate: bool, optional
784            Integrate over bandpass if True. Default: False.
785        return_alms: bool, optional
786            Return alms if True, else return maps. Default: False.
787        lmin: int, optional
788            Minimum multipole to keep (applies filtering).
789        coordinates: str, optional
790            Coordinate system for output ("G" (Galactic), "E" (Ecliptic), or "C" (Equatorial)). 
791            Default: "G"
792        **kwargs: Additional arguments passed to `hp.map2alm`.
793
794    Returns
795    -------
796        np.ndarray: 
797            Array of foreground maps or alms with shape (n_channels, 3, npix) for maps or (n_channels, 3, nalms) for alms.
798    """
799
800    fg_component = []
801
802    fell = _get_ell_filter(lmin, lmax) if lmin > 2 else None
803
804    rot = hp.Rotator(coord=f"G{coordinates}") if coordinates != "G" else None
805    
806    # Getting foreground component for each frequency channel
807    for idx, freq in enumerate(instrument.frequency):
808        if bandpass_integrate:
809            if hasattr(instrument, 'path_bandpasses'):
810                # Reading bandpass from file
811                bandpass_file = instrument.path_bandpasses + f"_{instrument.channels_tags[idx]}.npy"
812                frequencies, bandpass_weights = np.load(bandpass_file)
813                frequencies = frequencies * u.GHz
814            else:
815                # Create a top-hat bandpass
816                freq_min = freq * (1 - ( instrument.bandwidth[idx] / 2 ))
817                freq_max = freq * (1 + ( instrument.bandwidth[idx] / 2 ))
818                steps = int(freq_max - freq_min + 1)
819                frequencies = np.linspace(freq_min, freq_max, steps) * u.GHz
820                bandpass_weights = np.ones(len(frequencies)) # The tophat is defined in intensity units (Jy/sr)
821            with threadpool_limits(limits=1):
822                emission = sky.get_emission(frequencies, bandpass_weights)
823        else:
824            with threadpool_limits(limits=1):
825                emission = sky.get_emission(freq * u.GHz)
826
827        # Computing alms of the foreground emission
828        alm_emission = hp.map2alm(emission.value, lmax=lmax, pol=True, **kwargs)
829
830        # Applying coordinate rotation if needed
831        if coordinates != "G":
832            rot.rotate_alm(alm_emission, inplace=True)
833        
834        # Smoothing the alms with the instrument beam
835        if instrument.beams == "gaussian":
836            alm_emission = _smooth_input_alms_(
837                alm_emission,
838                fwhm=instrument.fwhm[idx],
839                nside_out=nside_in if pixel_window else None
840            )
841        else:
842            beamfile = instrument.path_beams + f"_{instrument.channels_tags[idx]}.fits"
843            alm_emission = _smooth_input_alms_(
844                alm_emission,
845                beam_path=beamfile,
846                symmetric_beam=(instrument.beams == "file_l"),
847                nside_out=nside_in if pixel_window else None
848            )
849
850        if lmin > 2:
851            for f in range(3):
852                alm_emission[f] = hp.almxfl(alm_emission[f], fell)
853        fg_component.append(alm_emission if return_alms else hp.alm2map(alm_emission, nside_out, lmax=lmax, pol=True))
854        
855    return np.array(fg_component)

857def _smooth_input_alms_(
858    alms: np.ndarray,
859    fwhm: Optional[float] = None,
860    nside_out: Optional[int] = None,
861    beam_path: Optional[str] = None,
862    symmetric_beam: bool = True
863) -> np.ndarray:
864    """
865    Apply beam and pixel window smoothing to input alms.
866
867    Parameters
868    ----------
869        alms: np.ndarray
870            Array of spherical harmonic coefficients [T, E, B].
871        fwhm: float, optional
872            FWHM of Gaussian beam in arcmin. Required if `beam_path` is None.
873        nside_out: int, optional
874            HEALPix Nside for pixel window function. Used if not None.
875        beam_path: str, optional
876            Path to FITS file containing beam transfer functions.
877        symmetric_beam: bool, optional
878            Whether the beam from FITS file is symmetric (True) or not (False).
879
880    Returns
881    -------
882        np.ndarray: 
883            Smoothed alms.
884    """
885    lmax = hp.Alm.getlmax(alms.shape[1])
886
887    # Beams transfer functions
888    if beam_path is not None:
889        bl_i = _get_beam_from_file(beam_path, lmax,symmetric_beam=symmetric_beam)
890    elif fwhm is not None:
891        symmetric_beam = True
892        bl_i = hp.gauss_beam(np.radians(fwhm/60.), lmax = lmax, pol = True)
893    else:
894        raise ValueError("Either fwhm or beam_path must be provided.")
895
896    bl_i = bl_i[:,:3]  # Take only T, E, B
897
898    if nside_out:
899        pw = hp.pixwin(nside_out, pol=True, lmax=lmax)
900        pw = np.array([pw[0], pw[1], pw[1]])
901        if symmetric_beam:
902            bl_i = bl_i * pw.T
903        else:
904            for i in range(3):
905                bl_i[:,i] = hp.almxfl(bl_i[:,i], pw[i])
906
907    # Initializing smoothed alms
908    alms_smoothed = np.zeros_like(alms)
909
910    for i in range(3):
911        alms_smoothed[i] = hp.almxfl(alms[i], bl_i[:,i]) if symmetric_beam else alms[i] * bl_i[:,i]
912
913    return alms_smoothed
