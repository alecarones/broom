 17def _compute_spectra(config: Configs) -> Optional[SimpleNamespace]:
 18
 19    """
 20    Compute the spectra from outputs of component separation.
 21    
 22    Parameters
 23    ----------
 24        config : Configs
 25            Configuration object containing parameters for spectra computation. It should include:
 26            - nsim_start: Starting simulation number.
 27            - nsims: Number of simulations to compute spectra for.
 28            - compute_spectra: List of dictionaries containing parameters for spectra computation, including:
 29                - path_method: Path to the component separation outputs.
 30                - field_out: Fields of the ouputs to be loaded for spectra computation.
 31                - components_for_cls: List of components to compute spectra for.
 32                - mask_type: Type of mask to apply (e.g., 'GAL*+fgres', 'GAL*0', 'GAL97', 'GAL99', 'fgres', 'config+fgres', 'config').
 33                - apodize_mask: Type of apodization to apply to the mask (e.g., 'gaussian', 'C1'). If None, no apodization is applied.
 34                - smooth_mask: Apodization scale for the mask in degrees. Default is 5 degrees.
 35                - nmt_purify_B: Whether to purify B modes when using NaMaster. Default is True.
 36                - nmt_purify_E: Whether to purify E modes when using NaMaster. Default is False.
 37                - smooth_tracer: Smoothing scale for the tracer in degrees, used when mask_type contains 'fgres'. Default is 3 degrees.
 38                - fsky: Fraction of the sky to consider when computing the spectra, used when mask_type contains 'fgres'. Default is 1.0.
 39            - spectra_comp: Method to compute the spectra, either 'anafast' or 'namaster'.
 40            - return_Dell: Whether to return the spectra as D_ell or C_ell. Default is False (returns C_ell).
 41            - field_cls_out: Fields of the spectra to compute. 
 42            - save_spectra: Whether to save the computed spectra to files. Default is True.
 43            - path_outputs: Path where computed spectra will be saved.
 44            - return_spectra: Whether to return the computed spectra as a SimpleNamespace object. Default is True.
 45    
 46    Returns
 47    -------
 48        cls_ : SimpleNamespace
 49            Object containing computed spectra with attributes for each component.
 50            Each attribute is a numpy array with dimensions (nsim, ncases, nfields, nbins), 
 51            where ncases is the number of different component separation outputs provided in compute_spectra dictionary.
 52    """
 53    
 54    if not isinstance(config, Configs):
 55        raise TypeError("config must be an instance of Configs")
 56
 57    cls_ = SimpleNamespace() if config.return_spectra else None
 58
 59    for nsim in range(config.nsim_start, config.nsim_start + config.nsims):
 60        if config.return_spectra:
 61            cls_sim = _compute_spectra_(config, nsim=nsim)
 62            for attr, value in vars(cls_sim).items():
 63                if not hasattr(cls_, attr):
 64                    setattr(cls_, attr, [])
 65                getattr(cls_, attr).append(value)
 66        else:
 67            _compute_spectra_(config, nsim=nsim)
 68
 69    if config.return_spectra:
 70        for attr in vars(cls_):
 71            setattr(cls_, attr, np.array(getattr(cls_, attr)))
 72        return cls_

 79    """
 80    Compute the spectra from outputs of component separation for single simulation 'nsim'.

 82    Parameters

 84        config : Configs
 85            Configuration object containing parameters for spectra computation. See `_compute_spectra` for details.
 86
 87        nsim : int or str, optional
 88            Simulation number to compute spectra for. If an integer, it will be zero-padded to 5 digits.
 89            Default is None, which means it will look for outputs with no label regarding simulation number.

 91    Returns
 92    -------
 93        cls_out_ : SimpleNamespace
 94            Object containing computed spectra with attributes for each component.

 98
 99    nsim = _format_nsim(nsim)

102        raise ValueError('spectra_comp must be either "anafast" or "namaster"')
103

115            _cls_from_config(config, compute_cls, nsim=nsim)

122def _standardize_compute_cls(config: Configs, compute_cls: Dict[str, Any]) -> Dict[str, Any]:
123    """
124    Standardize the compute_cls dictionary to ensure it contains all necessary fields and has the correct structure.
125
126    Parameters
127    ----------
128        config : Configs
129            Configuration object containing parameters for spectra computation. See `_compute_spectra` for details.
130        compute_cls : dict
131            Dictionary containing parameters for spectra computation. See `_compute_spectra` for details.
132    
133    Returns
134    -------
135        compute_cls : dict
136            Standardized dictionary containing parameters for spectra computation.
137    """
138
139    if 'path_method' not in compute_cls:
140        raise ValueError('compute_cls must contain a "path_method" field')
141
142    compute_cls.setdefault('field_out', config.field_out)
143
144    if "components_for_cls" not in compute_cls:
145        raise ValueError('compute_cls must contain a "components_for_cls" field')
146
147    compute_cls.setdefault("mask_type", None)
148    compute_cls.setdefault("apodize_mask", None)
149
150    if compute_cls["apodize_mask"] is not None:
151        compute_cls.setdefault("smooth_mask", 5.)
152
153    if compute_cls["mask_type"] is not None and "fgres" in compute_cls["mask_type"]:
154        compute_cls.setdefault("smooth_tracer", 3.0)
155        if "fsky" not in compute_cls:
156            compute_cls["fsky"] = 1.
157            print("'fsky' not defined in compute_spectra; set to 1.0 (no fg residual thresholding).")
158    if config.spectra_comp == 'namaster':
159        compute_cls.setdefault("nmt_purify_B", True)
160        compute_cls.setdefault("nmt_purify_E", False)
161
162    return compute_cls

176        compute_cls : dict

178        nsim : str, optional
179            Simulation number. Used to load the correct outputs. If None, it will look for outputs with no label regarding simulation number.
180    
181    Returns

185            Each attribute is a numpy array with dimensions (nfields, nbins),

228        save_spectra(config, cls_out, compute_cls, nsim=nsim)

230    # Clean up
231    compute_cls.pop("outputs", None)
232    compute_cls.pop("path", None)

299
300    bls_beam = get_bls(config.nside, config.fwhm_out, config.lmax, config.field_cls_out, pixel_window_out=config.pixel_window_out)

302    cls_out = SimpleNamespace()
303    for attr in vars(compute_cls["outputs"]):

311    def compute_anafast_scalar(maps, mask, beam):
312        cl = hp.anafast(maps * mask, lmax=config.lmax, pol=False)
313        cl /= np.mean(mask**2) * beam**2
314        if mask_in_maps is not None:
315            cl /= np.mean(mask_in_maps**2)
316        return b_bin.bin_cell(cl)
317
318    def compute_namaster_scalar(maps, mask, beam, field1):
319        f = nmt.NmtField(mask, [maps], beam=beam, lmax=config.lmax, lmax_mask=config.lmax)
320        if mask_in_maps is not None:
321            if f"00_{field1}{field1}" not in wspaces:
322                f_w = nmt.NmtField(mask * mask_in_maps, [maps], beam=beam, lmax=config.lmax, lmax_mask=config.lmax)
323                wspaces[f"00_{field1}{field1}"] = nmt.NmtWorkspace.from_fields(f_w, f_w, b_bin)
324        else:
325            if f"00_{field1}{field1}" not in wspaces:
326                wspaces[f"00_{field1}{field1}"] = nmt.NmtWorkspace.from_fields(f, f, b_bin)
327        return (nmt.compute_full_master(f, f, b_bin, workspace=wspaces[f"00_{field1}{field1}"]))[0]
328        
329    def compute_anafast_cross_scalars(map1, map2, mask, beam1, beam2):
330        cl_cross = hp.anafast(map1 * mask, map2=map2 * mask,lmax=config.lmax,pol=False)
331        cl_cross /= np.mean(mask**2) * (beam1 * beam2)
332        if mask_in_maps is not None:
333            cl_cross /= np.mean(mask_in_maps**2)
334        return b_bin.bin_cell(cl_cross)

336    def compute_namaster_cross_scalars(map1, map2, mask1, mask2, beam1, beam2, field1, field2):

338        f2 = nmt.NmtField(mask2, [map2],beam=beam2,lmax=config.lmax,lmax_mask=config.lmax)
339        if mask_in_maps is not None:
340            f1_w = nmt.NmtField(mask1 * mask_in_maps, [map1],beam=beam1,lmax=config.lmax,lmax_mask=config.lmax)
341            f2_w = nmt.NmtField(mask2 * mask_in_maps, [map2],beam=beam2,lmax=config.lmax,lmax_mask=config.lmax)
342            if f"00_{field1}{field2}" not in wspaces:
343                wspaces[f"00_{field1}{field2}"] = nmt.NmtWorkspace.from_fields(f1_w, f2_w, b_bin)        
344        else:
345            if f"00_{field1}{field2}" not in wspaces:
346                wspaces[f"00_{field1}{field2}"] = nmt.NmtWorkspace.from_fields(f1, f2, b_bin)
347        return (nmt.compute_full_master(f1, f2, b_bin, workspace=wspaces[f"00_{field1}{field2}"]))[0]

349    def compute_anafast_full_TQU(T_map, Q_map, U_map, T_mask, Q_mask, beam_T, beam_E, beam_B, get_cross=False):

351        cl[0] = cl[0] / np.mean(T_mask**2) / (beam_T**2)
352        cl[1] = cl[1] / np.mean(Q_mask**2) / (beam_E**2)
353        cl[2] = cl[2] / np.mean(Q_mask**2) / (beam_B**2)
354        if mask_in_maps is not None:
355            cl /= np.mean(mask_in_maps**2)
356        if get_cross:
357            mask_ =  Q_mask if np.mean(np.ceil(T_mask)) > np.mean(np.ceil(Q_mask)) else T_mask
358            cl_cross = (hp.anafast([T_map * mask_, Q_map * mask_, U_map * mask_], lmax=config.lmax, pol=True) / np.mean(mask_**2))[3:]
359            cl_cross[0] = cl_cross[0] / (beam_T * beam_E)
360            cl_cross[1] = cl_cross[1] / (beam_E * beam_B)
361            cl_cross[2] = cl_cross[2] / (beam_T * beam_B)
362            if mask_in_maps is not None:
363                cl_cross /= np.mean(mask_in_maps**2)
364            return np.concatenate([b_bin.bin_cell(cl), b_bin.bin_cell(cl_cross)], axis=0)
365        return b_bin.bin_cell(cl)
