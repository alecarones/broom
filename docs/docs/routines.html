  9def _get_ell_filter(lmin: int, lmax: int) -> np.ndarray:
 10    """
 11    Return a filter array in harmonic space with filtering at low multipoles.
 12    
 13    Parameters
 14    ----------
 15        lmin : int
 16            Minimum multipole to consider.
 17        lmax : int
 18            Maximum multipole to consider.
 19    
 20    Returns
 21    -------
 22        fell : np.ndarray
 23            Filter array of shape (lmax+1,) with values set to 1 for multipoles >= lmin,
 24            and 0 for multipoles < lmin, with a cosine transition for low multipoles.
 25    """
 26    fell = np.ones(lmax+1)
 27    if lmin <=3:
 28        fell[:lmin] = 0.
 29    elif lmin <= 6:
 30        fell[:2]=0.
 31        fell[2:lmin+1]=np.cos((lmin - np.arange(2,lmin+1)) * np.pi / (2. * (lmin-2)))
 32    else:
 33        fell[:lmin-4]=0.
 34        fell[lmin-4:lmin+1]=np.cos((lmin - np.arange(lmin-4,lmin+1)) * np.pi / 8.)
 35    return fell

121def _bl_from_fwhms(fwhm_out: float, fwhm_in: float, lmax: int) -> np.ndarray:
122    """
123    Compute beam transfer function to correct for input gaussian beam and convolve for output beam.

144def _bl_from_file(beam_path: str, channel: str, fwhm_out: float, input_beams: str, lmax: int) -> np.ndarray:
145    """
146    Load beam transfer function from file and compute output/input ratio.
147    
148    Parameters
149    ----------
150        beam_path : str
151            Path to the beam transfer function file.
152        channel : str
153            Channel identifier for the beam file.
154        fwhm_out : float
155            Full width at half maximum of the output beam in arcminutes.
156        input_beams : str
157            Type of input beams, either "file_l" or "file_lm".
158        lmax : int
159            Maximum multipole to consider for the beam transfer function.
160
161    Returns
162    -------
163        np.ndarray
164            Beam transfer function of shape (lmax+1, 3) for T, E, B modes.
165    """
166    beam_file = f"{beam_path}_{channel}.fits"
167    
168    bl_in = _get_beam_from_file(beam_file,lmax,symmetric_beam=(input_beams != "file_lm"))
169    bl_out_l = hp.gauss_beam(np.radians(fwhm_out/60.), lmax=lmax,pol=True)
170
171    if input_beams == "file_l":
172        return (bl_out_l[:,:3]) / (bl_in[:,:3])
173    elif input_beams == "file_lm":
174        bl_out = np.zeros((hp.Alm.getsize(lmax),3),dtype=complex)
175        for ell in range(0, lmax + 1):
176            idx_lmax = np.array([hp.Alm.getidx(lmax, ell, m) for m in range(ell + 1)])
177            bl_out[idx_lmax,:] = np.tile(bl_out_l[ell,:3], (ell+1, 1))
178        return bl_out / (bl_in[:,:3])

180def  _get_beam_from_file(beam_file: str, lmax: int, symmetric_beam: bool = True) -> np.ndarray:
181    """
182    Read beam transfer function from FITS file.
183    
184    Parameters
185    ----------
186        beam_file : str
187            Path to the beam transfer function file.
188        lmax : int
189            Maximum multipole to consider for the beam transfer function.
190        symmetric_beam : bool, optional
191            If True, assume the beam is symmetric. Default is True.
192
193    Returns
194    -------
195        np.ndarray
196            Beam transfer function of shape (lmax+1, 3) for T, E, B modes.
197    
198    """
199    hdul = fits.open(beam_file)
200    bl_file = np.column_stack([hdul[1].data[col].astype(str).astype(float) for col in hdul[1].data.names]).squeeze()
201
202    if bl_file.ndim != 2 or bl_file.shape[1] not in [3, 4]:
203        raise ValueError("Beam file must be 2-dimensional and have 3 or 4 columns (for T, E, B and EB).")
204
205    if symmetric_beam:
206        if lmax >= bl_file.shape[0]:
207            raise ValueError("Insufficient multipoles in beam file.")
208        return bl_file[:lmax+1]
209
210    if lmax > hp.Alm.getlmax(bl_file.shape[0]):
211        raise ValueError("The provided asymmetric beam file does not have enough values for the given alm.")
212
213    bl = np.zeros((hp.Alm.getsize(lmax),bl_file.shape[1]),dtype=complex)
214    lmax_file = hp.Alm.getlmax(bl_file.shape[0])
215    for ell in range(0, lmax + 1):
216        idx_lmax = np.array([hp.Alm.getidx(lmax, ell, m) for m in range(ell + 1)])
217        idx_lmax_file = np.array([hp.Alm.getidx(lmax_file, ell, m) for m in range(ell + 1)])
218        bl[idx_lmax,:] = bl_file[idx_lmax_file,:]
219    return bl

251def _get_local_cov(
252    input_maps: np.ndarray,
253    lmax: int,
254    ilc_bias: float,
255    b_ell: Optional[np.ndarray] = None,
256    mask: Optional[np.ndarray] = None,
257    reduce_bias: bool = False,
258    input_maps_2: Optional[np.ndarray] = None
259) -> np.ndarray:
260    """
261    Compute the local covariance matrix between input maps.
262
263    Parameters
264    ----------
265        input_maps: np.ndarray
266            Input maps of shape (n_channels, n_pixels).
267        lmax: int 
268            Maximum multipole used in the harmonic space.
269        ilc_bias: float
270            Term used for smoothing scale estimation to optimally sample the covariance.
271        b_ell: np.ndarray, optional
272            Needlet transfer function. Defaults: unity.
273        mask: np.ndarray, optional
274            Optional sky mask. If provided, the covariance is computed only on the non-masked pixels.
275        reduce_bias: bool
276            If True, apply ILC bias-reduction technique.
277        input_maps_2: np.ndarray, optional
278            Second set of maps for computing cross-covariance.
279
280    Returns
281    -------
282        cov: np.ndarray
283            Covariance matrices of shape (n_channels, n_channels, n_pixels).
284    """
285    if not isinstance(b_ell, np.ndarray):
286        b_ell = np.ones(lmax+1)
287
288    nmodes_band = np.sum((2. * np.arange(lmax + 1) + 1.) * (b_ell ** 2))
289    pps = np.sqrt(float(input_maps.shape[1]) * float(input_maps.shape[0]-1) / (ilc_bias * nmodes_band) )
290
291    n_channels = input_maps.shape[0]
292    nside_cov = hp.npix2nside(input_maps.shape[1]) if mask is not None else int(np.min([hp.npix2nside(input_maps.shape[1]), 128]))
293
294    cov = np.zeros((n_channels, n_channels, 12 * nside_cov ** 2))
295
296    for i in range(n_channels):
297        for k in range(i,n_channels):
298            map1 = input_maps[i] * mask if mask is not None else input_maps[i]
299            map2 = (input_maps_2[k] if input_maps_2 is not None else input_maps[k])
300            map2 = map2 * mask if mask is not None else map2
301
302            cov[i,k] = cov_map = _get_local_cov_(
303                map1, map2, pps, nside_cov, reduce_bias=reduce_bias
304            )
305
306    for i in range(n_channels):
307        for k in range(i):
308            cov[i,k]=cov[k,i]
309            
310    return cov

312def _get_local_cov_(
313    map1: np.ndarray,
314    map2: np.ndarray,
315    pps: float,
316    nside_covar: Optional[int] = None,
317    reduce_bias: bool = False
318) -> np.ndarray:
319    """
320    Compute local covariance between two maps.
321
322    Parameters
323    ----------
324        map1: np.ndarray
325            First input map.
326        map2: np.ndarray
327            Second input map.
328        pps: float
329            Smoothing factor.
330        nside_covar: int, optional
331            Nside resolution of output covariance. If None, it is set to Nside of map1 and map2
332        reduce_bias: bool
333            Apply smoothing bias reduction if True.
334
335    Returns
336    -------
337        np.ndarray
338            Covariance.
339    """
340    if hp.get_nside(map1) != hp.get_nside(map2):
341        raise ValueError("Input maps must have the same HEALPix resolution.")
342
343    product_map = map1 * map2
344    npix = product_map.size
345    nside = hp.get_nside(product_map)
346    if nside_covar is None:
347        nside_covar = nside
348
349    # Degrade resolution for faster smoothing
350    nside_out = max(1, nside // 4)
351    stat_map = hp.ud_grade(product_map, nside_out = nside_out, order_in = 'RING', order_out = 'RING')
352    
353    # Compute alm
354    lmax_stat = 2 * nside_out #3 * nside_out - 1 # 
355    alm_s = hp.map2alm(stat_map, lmax=lmax_stat, iter=1, use_weights=True)# iter=0?
356
357    # Get beam for smoothing covariance
358    pix_size = np.sqrt(4.0 * np.pi / npix)
359    fwhm_stat = pps * pix_size
360    bl_stat = hp.gauss_beam(fwhm_stat, lmax_stat)
361
362    # If reduce_bias is True, apply bias reduction technique
363    if reduce_bias:
364        thetas = np.arange(0,np.pi,0.002)
365        beam_stat = hp.bl2beam(bl_stat, thetas)
366        theta_ = 0.5 * fwhm_stat
367        dist = 0.5 * (np.tanh(30 * (thetas - 0.3 * theta_)) - np.tanh(30 * (thetas - 3 * theta_)))
368        dist /= np.max(dist)
369        dist[np.argmax(dist):]=1.
370        bl_stat = hp.beam2bl(dist * beam_stat,thetas, lmax = lmax_stat)
371
372    alm_s = hp.almxfl(alm_s, bl_stat)
373    
374    return hp.alm2map(alm_s, nside_covar, lmax=lmax_stat)

571    """
572    Slice data fields from a SimpleNamespace object according to field_in and field_out.
573
574    Parameters
575    ----------
